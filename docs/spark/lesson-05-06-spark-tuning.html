§ 01
DataFrame API vs Spark SQL
同一逻辑，两种写法——性能完全相同，选择看场景
api_style.py 
from pyspark.sql import SparkSession
from pyspark.sql.functions import col, count, sum, avg, round, when

# ── 电商场景：统计每类商品的 GMV、订单量、转化率 ──

result = orders_df \
    .filter(col("status") != "cancelled") \
    .join(products_df, on="product_id", how="left") \
    .groupBy("category") \
    .agg(
        count("order_id").alias("order_cnt"),
        sum("amount").alias("gmv"),
        round(avg("amount"), 2).alias("avg_order"),
        sum(when(col("is_paid") == 1, 1).otherwise(0)).alias("paid_cnt")
    ) \
    .withColumn("pay_rate",
        round(col("paid_cnt") / col("order_cnt") * 100, 1)) \
    .orderBy(col("gmv").desc())

result.show(20, truncate=False)
sql_style.py 
# 先注册视图
orders_df.createOrReplaceTempView("orders")
products_df.createOrReplaceTempView("products")

# ── 完全相同的逻辑，SQL 写法 ──
result = spark.sql("""
    SELECT
        p.category,
        COUNT(o.order_id)                        AS order_cnt,
        SUM(o.amount)                            AS gmv,
        ROUND(AVG(o.amount), 2)                  AS avg_order,
        SUM(CASE WHEN o.is_paid = 1 THEN 1
                 ELSE 0 END)                     AS paid_cnt,
        ROUND(
            SUM(CASE WHEN o.is_paid = 1 THEN 1
                     ELSE 0 END)
            / COUNT(o.order_id) * 100, 1
        )                                        AS pay_rate
    FROM orders o
    LEFT JOIN products p ON o.product_id = p.product_id
    WHERE o.status != 'cancelled'
    GROUP BY p.category
    ORDER BY gmv DESC
""")

result.show(20, truncate=False)

HIVE SQL（原版） 慢·T+1 
SELECT
    DATE_FORMAT(create_time, 'yyyy-MM') AS mon,
    category,
    SUM(amount) AS gmv,
    COUNT(DISTINCT user_id) AS buyers
FROM dwd.order_detail

GROUP BY 1, 2

SPARK SQL（重写） 快·内存 
# 读 Parquet（比 Hive ORC 更通用）
df = spark.read.parquet("oss://bucket/orders/")
df.createOrReplaceTempView("orders")

spark.sql("""
    SELECT
        date_format(create_time,'yyyy-MM') AS mon,
        category,
        SUM(amount)              AS gmv,
        COUNT(DISTINCT user_id)  AS buyers
    FROM orders
    WHERE create_time >= '2024-01-01'
      AND status = 'paid'
    GROUP BY 1, 2
    ORDER BY mon, gmv DESC
""").show()
💡 迁移要点：Hive 的 DATE_FORMAT → Spark SQL 同名函数直接用；表路径从 Hive Metastore 改成直接读文件；分区字段 dt 改成用 create_time 过滤（Spark 自动 partition pruning）
retention.py  · 复购率：30天内下过≥2单的用户比例 
spark.sql("""
    WITH user_orders AS (
        SELECT
            user_id,
            COUNT(DISTINCT order_id)  AS order_cnt,
            MIN(create_time)          AS first_order,
            MAX(create_time)          AS last_order
        FROM orders
        WHERE status = 'paid'
          AND create_time >= date_sub(current_date(), 30)
        GROUP BY user_id
    ),
    labeled AS (
        SELECT *,
            CASE WHEN order_cnt >= 2 THEN 1 ELSE 0 END AS is_repeat
        FROM user_orders
    )
    SELECT
        SUM(is_repeat)                               AS repeat_buyers,
        COUNT(*)                                     AS total_buyers,
        ROUND(SUM(is_repeat) / COUNT(*) * 100, 2)   AS repeat_rate_pct
    FROM labeled
""").show()
sales_rank.py  · 每品类 TOP3 销售额商品 
spark.sql("""
    WITH category_sales AS (
        SELECT
            p.category,
            p.product_name,
            SUM(o.amount) AS sales
        FROM orders o
        JOIN products p ON o.product_id = p.product_id
        WHERE o.status = 'paid'
        GROUP BY p.category, p.product_name
    ),
    ranked AS (
        SELECT *,
            RANK() OVER (
                PARTITION BY category
                ORDER BY sales DESC
            ) AS rnk
        FROM category_sales
    )
    SELECT category, product_name, sales, rnk
    FROM ranked
    WHERE rnk <= 3
    ORDER BY category, rnk
""").show(50, truncate=False)

# 用户行为日志已注册为 user_events 视图
# schema: user_id, event_type, product_id, ts

spark.sql("""
    WITH step_flags AS (
        SELECT
            user_id,
            MAX(CASE WHEN event_type = 'expose'   THEN 1 ELSE 0 END) AS s1_expose,
            MAX(CASE WHEN event_type = 'click'    THEN 1 ELSE 0 END) AS s2_click,
            MAX(CASE WHEN event_type = 'add_cart' THEN 1 ELSE 0 END) AS s3_cart,
            MAX(CASE WHEN event_type = 'order'    THEN 1 ELSE 0 END) AS s4_order,
            MAX(CASE WHEN event_type = 'pay'      THEN 1 ELSE 0 END) AS s5_pay
        FROM user_events
        WHERE dt = '2024-06-01'
        GROUP BY user_id
    ),
    funnel_counts AS (
        SELECT
            SUM(s1_expose)                           AS expose_uv,
            SUM(s1_expose * s2_click)                AS click_uv,
            SUM(s1_expose * s2_click * s3_cart)      AS cart_uv,
            SUM(s1_expose * s2_click * s3_cart
                * s4_order)                          AS order_uv,
            SUM(s1_expose * s2_click * s3_cart
                * s4_order * s5_pay)                 AS pay_uv
        FROM step_flags
    )
    SELECT
        expose_uv,  click_uv,  cart_uv,  order_uv,  pay_uv,
        ROUND(click_uv / expose_uv * 100, 2)         AS expose2click,
        ROUND(cart_uv  / click_uv  * 100, 2)         AS click2cart,
        ROUND(order_uv / cart_uv   * 100, 2)         AS cart2order,
        ROUND(pay_uv   / order_uv  * 100, 2)         AS order2pay,
        ROUND(pay_uv   / expose_uv * 100, 2)         AS overall_cvr
    FROM funnel_counts
""").show()

§ 01
数据倾斜：原因与定位
为什么99个Task都完成了，第100个还在跑？
💀
什么是数据倾斜
GroupBy/Join 后，某个 key 的数据量远大于其他 key，导致一个 Task 处理了 90% 的数据，其他 Task 早就空闲了
99 tasks: 2s ｜ 1 task: 47min
🔍
如何定位
Spark UI → Stages → 找 max duration 远大于 median 的 Stage；看 Task 的 Shuffle Read Size 分布
Spark UI → Jobs → Stages → Tasks
📊
典型触发场景
电商：用户行为日志里 "爬虫账号" 产生百万条记录；Join 时 NULL 值被当成同一个 key 聚合
SELECT count(*) GROUP BY user_id → 找出异常 key
TASK DURATION · BEFORE FIX
median: 1.8s max: 47min
DIAGNOSIS CODE
find_skew.py
# Step 1：找出哪个 key 数据量异常
df.groupBy("user_id") \
  .count() \
  .orderBy(col("count").desc()) \
  .show(20)

# Step 2：统计 key 分布
df.groupBy("user_id") \
  .count() \
  .selectExpr(
      "percentile_approx(count, 0.5) as median",
      "percentile_approx(count, 0.99) as p99",
      "max(count) as max_count"
  ).show()

# Step 3：NULL key 检查（NULL 在 JOIN 时全聚到一个 Task）
df.filter(col("user_id").isNull()).count()
§ 02
解决方案：加盐 + 广播 Join
两种武器，对应不同场景
倾斜 key
user_id=BOT
→
加随机盐
BOT_0 ~ BOT_9
→
分散到10个Task
局部聚合
→
去盐再聚合
全局合并
salt_groupby.py
from pyspark.sql.functions import col, floor, rand, concat_ws, sum

SALT_NUM = 10  # 把倾斜 key 打散成10份

# Step 1：打盐 —— 给每行加随机后缀
salted = df.withColumn(
    "salted_key",
    concat_ws("_", col("user_id"), (floor(rand() * SALT_NUM)).cast("string"))
)

# Step 2：局部聚合（此时 key 已分散，无倾斜）
partial = salted.groupBy("salted_key") \
    .agg(sum("amount").alias("partial_sum"))

# Step 3：去盐 —— 截掉后缀，还原原始 key
partial = partial.withColumn(
    "user_id",
    col("salted_key").substr(1, col("salted_key").instr("_") - 1)
    # 或用 split(col("salted_key"),"_")[0]
)

# Step 4：全局再聚合（数量少，速度快）
result = partial.groupBy("user_id") \
    .agg(sum("partial_sum").alias("total_amount"))

result.show()
大表 Join 小表
触发 Shuffle
→
broadcast(小表)
发到每个节点
→
各节点本地 Join

broadcast_join.py
from pyspark.sql.functions import broadcast

# 场景：orders（100亿行）JOIN users（5000行）
# 用户表很小，直接广播

result = orders_df.join(
    broadcast(users_df),  # ← 关键：把小表标记为广播
    on="user_id",
    how="left"
)

# ── 自动广播阈值（默认 10MB，可调大）──
spark.conf.set("spark.sql.autoBroadcastJoinThreshold", "50m")

# ── 两张大表都有倾斜 key？先过滤倾斜部分分别处理 ──
SKEW_KEYS = ["BOT_001", "BOT_002"]

normal = orders_df.filter(~col("user_id").isin(SKEW_KEYS))
skewed = orders_df.filter( col("user_id").isin(SKEW_KEYS))

result_normal = normal.join(users_df,  on="user_id", how="left")
result_skewed = skewed.join(broadcast(users_df.filter(
    col("user_id").isin(SKEW_KEYS)
)), on="user_id", how="left")

final = result_normal.union(result_skewed)

spark_config_template.py  · 生产配置模板
spark = SparkSession.builder \
    .appName("prod-job") \
    # ── Executor 资源 ──
    .config("spark.executor.memory",         "8g")     # 每个 Executor 内存
    .config("spark.executor.cores",           "4")      # 每个 Executor CPU核
    .config("spark.executor.instances",       "20")     # Executor 数量
    .config("spark.driver.memory",            "4g")     # Driver 内存
    # ── 并行度 ──
    .config("spark.default.parallelism",       "160")    # RDD 默认并行度
    .config("spark.sql.shuffle.partitions",    "200")    # Shuffle 分区数（最重要）
    # ── 广播 ──
    .config("spark.sql.autoBroadcastJoinThreshold", "20971520")  # 20MB
    # ── 动态资源 ──
    .config("spark.dynamicAllocation.enabled",  "true")
    .config("spark.dynamicAllocation.minExecutors", "2")
    .config("spark.dynamicAllocation.maxExecutors", "50")
    # ── AQE（Spark 3.x 自适应查询执行）──
    .config("spark.sql.adaptive.enabled",       "true")   # 强烈推荐开启
    .config("spark.sql.adaptive.skewJoin.enabled","true")  # 自动处理倾斜
    .getOrCreate()
参数默认值说明 & 调优建议
spark.sql.shuffle.partitions 200 Shuffle 后的分区数。最常调的参数。数据量大时调大（如1000），数据量小时调小（如50）防止产生大量空 Task公式参考：数据量(GB) × 2 ~ 4
spark.executor.memory 1g 每个 Executor 的内存。OOM 时调大；但单个 Executor 不宜超过 64g（GC 压力大）生产推荐：4g ~ 16g
spark.executor.cores 1 每个 Executor 的 CPU 核数。影响并发 Task 数。推荐 2~5，过大会导致内存竞争Task 并发 = cores × instances
spark.sql.adaptive.enabled true (3.x) AQE 自适应查询执行：运行时动态调整 shuffle 分区数、处理倾斜。Spark 3.0+ 默认开启，强烈保持 true开启后可省去大量手动调参
spark.memory.fraction 0.6 Spark 内存池占堆内存的比例（用于执行和存储）。遇到 OOM 可适当调大到 0.75
spark.sql.autoBroadcastJoinThreshold 10MB 小于此值的表自动广播。设为 -1 禁用自动广播。可按实际小表大小调整推荐：20m ~ 100m
