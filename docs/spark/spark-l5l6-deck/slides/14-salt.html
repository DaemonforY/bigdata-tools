SALT_NUM = 10 # Step1 打盐 salted = df.withColumn("salted_key", concat_ws("_", col("user_id"), floor(rand()*SALT_NUM).cast("string"))) # Step2 局部聚合（key 已分散，无倾斜） partial = salted.groupBy("salted_key").agg(sum("amount").alias("partial_sum")) # Step3 去盐 + Step4 全局再聚合 partial = partial.withColumn("user_id", split(col("salted_key"),"_")[0]) result = partial.groupBy("user_id").agg(sum("partial_sum").alias("total_amount"))
